import { useEffect, useState, FormEvent } from 'react'
import { useNavigate } from 'react-router-dom'
import { useAuth } from '@/hooks/useAuth'
import { useAuthStore } from '@/context/authStore'
import { apiClient } from '@/services/api'
import { User } from '@/types/auth'
import { AdminMetrics } from '@/types/admin'
import {
  Loader2,
  Shield,
  Users,
  FileText,
  Github,
  AlertTriangle,
  Plus,
  Search,
  Settings,
  BarChart3,
} from 'lucide-react'

type TabType = 'dashboard' | 'users' | 'repositories' | 'proposals' | 'issues' | 'settings'

const userLevels = ['ANONYMOUS', 'REGISTERED', 'VERIFIED', 'SPECIAL']

export function AdminPage() {
  const { user } = useAuth()
  const navigate = useNavigate()
  const { isLoading } = useAuthStore()
  const [activeTab, setActiveTab] = useState<TabType>('dashboard')
  const [metrics, setMetrics] = useState<AdminMetrics | null>(null)
  const [users, setUsers] = useState<User[]>([])
  const [loadingUsers, setLoadingUsers] = useState(true)
  const [savingUserId, setSavingUserId] = useState<number | null>(null)
  const [edits, setEdits] = useState<Record<number, Partial<User>>>({})
  const [error, setError] = useState<string | null>(null)
  const [success, setSuccess] = useState<string | null>(null)
  const [searchTerm, setSearchTerm] = useState('')
  const [showNewUserForm, setShowNewUserForm] = useState(false)
  const [newUserForm, setNewUserForm] = useState({
    username: '',
    email: '',
    password: '',
    full_name: '',
  })

  useEffect(() => {
    if (!isLoading && (!user || !user.is_superuser)) {
      navigate('/')
      return
    }

    const fetchData = async () => {
      try {
        const [metricsData, usersData] = await Promise.all([
          apiClient.getAdminMetrics(),
          apiClient.getUsers(),
        ])
        setMetrics(metricsData)
        setUsers(usersData)
      } catch (err) {
        console.error('Erro ao carregar painel admin', err)
        setError('Não foi possível carregar as informações do painel.')
      } finally {
        setLoadingUsers(false)
      }
    }

    if (user?.is_superuser) {
      fetchData()
    }
  }, [user, isLoading, navigate])

  const handleEditChange = (userId: number, field: keyof User, value: any) => {
    setEdits((prev) => ({
      ...prev,
      [userId]: {
        ...prev[userId],
        [field]: value,
      },
    }))
  }

  const handleSaveUser = async (userId: number) => {
    const payload = edits[userId]
    if (!payload || Object.keys(payload).length === 0) {
      return
    }

    setSavingUserId(userId)
    try {
      const updated = await apiClient.updateUser(userId, payload)
      setUsers((prev) => prev.map((u) => (u.id === userId ? updated : u)))
      setEdits((prev) => {
        const next = { ...prev }
        delete next[userId]
        return next
      })
      setSuccess('Usuário atualizado com sucesso!')
      setTimeout(() => setSuccess(null), 3000)
    } catch (err) {
      console.error('Erro ao atualizar usuário', err)
      setError('Não foi possível atualizar o usuário.')
    } finally {
      setSavingUserId(null)
    }
  }

  const handleNewUserSubmit = async (e: FormEvent) => {
    e.preventDefault()
    if (!newUserForm.username || !newUserForm.email || !newUserForm.password) {
      setError('Por favor, preencha todos os campos obrigatórios')
      return
    }

    try {
      const created = await apiClient.createUser({
        username: newUserForm.username,
        email: newUserForm.email,
        password: newUserForm.password,
        full_name: newUserForm.full_name || undefined,
      })
      setUsers((prev) => [created, ...prev])
      setNewUserForm({ username: '', email: '', password: '', full_name: '' })
      setShowNewUserForm(false)
      setSuccess('Usuário criado com sucesso!')
      setTimeout(() => setSuccess(null), 3000)
    } catch (err: any) {
      console.error('Erro ao criar usuário', err)
      const errorMsg = err.response?.data?.detail || 'Erro ao criar usuário'
      setError(errorMsg)
    }
  }

  const filteredUsers = users.filter(
    (u) =>
      u.username.toLowerCase().includes(searchTerm.toLowerCase()) ||
      u.email.toLowerCase().includes(searchTerm.toLowerCase()) ||
      (u.full_name?.toLowerCase().includes(searchTerm.toLowerCase()) ?? false)
  )

  if (isLoading || !user || !user.is_superuser) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <Loader2 className="w-10 h-10 text-primary-600 animate-spin" />
      </div>
    )
  }

  return (
    <div className="min-h-screen bg-gray-50 pb-16">
      <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-10 space-y-8">
        {/* Header */}
        <div className="flex items-center gap-3">
          <Shield className="w-8 h-8 text-primary-600" />
          <div>
            <h1 className="text-3xl font-bold text-gray-900">Área Administrativa</h1>
            <p className="text-gray-600 text-sm">Gerencie usuários, repositórios e configure o sistema.</p>
          </div>
        </div>

        {/* Messages */}
        {error && (
          <div className="rounded border border-red-200 bg-red-50 text-red-700 p-3 text-sm flex items-center gap-2">
            <AlertTriangle className="w-4 h-4" />
            {error}
          </div>
        )}
        {success && (
          <div className="rounded border border-green-200 bg-green-50 text-green-700 p-3 text-sm flex items-center gap-2">
            ✓ {success}
          </div>
        )}

        {/* Tabs Navigation */}
        <div className="border-b border-gray-200">
          <nav className="flex gap-8 -mb-px">
            {[
              { id: 'dashboard', label: 'Dashboard', icon: BarChart3 },
              { id: 'users', label: 'Usuários', icon: Users },
              { id: 'repositories', label: 'Repositórios', icon: Github },
              { id: 'proposals', label: 'Propostas', icon: FileText },
              { id: 'issues', label: 'Issues', icon: AlertTriangle },
              { id: 'settings', label: 'Configurações', icon: Settings },
            ].map(({ id, label, icon: Icon }) => (
              <button
                key={id}
                onClick={() => setActiveTab(id as TabType)}
                className={`pb-3 px-1 text-sm font-medium border-b-2 transition-colors ${
                  activeTab === id
                    ? 'border-primary-600 text-primary-600'
                    : 'border-transparent text-gray-600 hover:text-gray-900'
                } flex items-center gap-2`}
              >
                <Icon className="w-4 h-4" />
                {label}
              </button>
            ))}
          </nav>
        </div>

        {/* Tab Content */}
        {activeTab === 'dashboard' && (
          <section className="grid grid-cols-1 md:grid-cols-4 gap-6">
            <StatCard icon={<Github className="w-6 h-6" />} label="Repositórios" value={metrics?.repositories ?? 0} />
            <StatCard icon={<FileText className="w-6 h-6" />} label="Propostas" value={metrics?.proposals ?? 0} />
            <StatCard icon={<AlertTriangle className="w-6 h-6" />} label="Demandas" value={metrics?.issues ?? 0} />
            <StatCard icon={<Users className="w-6 h-6" />} label="Usuários" value={metrics?.users ?? 0} />
          </section>
        )}

        {activeTab === 'users' && (
          <section className="space-y-6">
            <div className="flex justify-between items-center">
              <div className="relative flex-1 max-w-md">
                <Search className="absolute left-3 top-3 w-5 h-5 text-gray-400" />
                <input
                  type="text"
                  placeholder="Buscar usuários..."
                  value={searchTerm}
                  onChange={(e) => setSearchTerm(e.target.value)}
                  className="input-field pl-10 w-full"
                />
              </div>
              <button onClick={() => setShowNewUserForm(!showNewUserForm)} className="btn-primary flex items-center gap-2">
                <Plus className="w-4 h-4" />
                Novo Usuário
              </button>
            </div>

            {showNewUserForm && (
              <div className="card space-y-4 bg-blue-50 border border-blue-200">
                <h3 className="font-semibold text-gray-900">Cadastrar Novo Usuário</h3>
                <form onSubmit={handleNewUserSubmit} className="grid grid-cols-1 md:grid-cols-2 gap-4">
                  <input
                    type="text"
                    placeholder="Username"
                    value={newUserForm.username}
                    onChange={(e) => setNewUserForm({ ...newUserForm, username: e.target.value })}
                    className="input-field"
                    required
                  />
                  <input
                    type="email"
                    placeholder="Email"
                    value={newUserForm.email}
                    onChange={(e) => setNewUserForm({ ...newUserForm, email: e.target.value })}
                    className="input-field"
                    required
                  />
                  <input
                    type="password"
                    placeholder="Senha"
                    value={newUserForm.password}
                    onChange={(e) => setNewUserForm({ ...newUserForm, password: e.target.value })}
                    className="input-field"
                    required
                  />
                  <input
                    type="text"
                    placeholder="Nome completo"
                    value={newUserForm.full_name}
                    onChange={(e) => setNewUserForm({ ...newUserForm, full_name: e.target.value })}
                    className="input-field"
                  />
                  <div className="flex gap-2">
                    <button type="submit" className="btn-primary">
                      Criar Usuário
                    </button>
                    <button type="button" onClick={() => setShowNewUserForm(false)} className="btn-outline">
                      Cancelar
                    </button>
                  </div>
                </form>
              </div>
            )}

            <div className="card">
              <h3 className="font-semibold mb-4 text-gray-900">Usuários ({filteredUsers.length})</h3>
              {loadingUsers ? (
                <div className="flex justify-center py-10">
                  <Loader2 className="w-10 h-10 text-primary-600 animate-spin" />
                </div>
              ) : (
                <div className="overflow-x-auto">
                  <table className="min-w-full divide-y divide-gray-200">
                    <thead className="bg-gray-50">
                      <tr>
                        <th className="px-4 py-2 text-left text-xs font-medium text-gray-500 uppercase">Usuário</th>
                        <th className="px-4 py-2 text-left text-xs font-medium text-gray-500 uppercase">Email</th>
                        <th className="px-4 py-2 text-left text-xs font-medium text-gray-500 uppercase">Nível</th>
                        <th className="px-4 py-2 text-left text-xs font-medium text-gray-500 uppercase">Admin</th>
                        <th className="px-4 py-2 text-left text-xs font-medium text-gray-500 uppercase">Status</th>
                        <th className="px-4 py-2 text-left text-xs font-medium text-gray-500 uppercase">Ações</th>
                      </tr>
                    </thead>
                    <tbody className="bg-white divide-y divide-gray-200">
                      {filteredUsers.map((u) => {
                        const pending = edits[u.id] || {}
                        return (
                          <tr key={u.id}>
                            <td className="px-4 py-3 text-sm text-gray-900 font-medium">{u.username}</td>
                            <td className="px-4 py-3 text-sm text-gray-500">{u.email}</td>
                            <td className="px-4 py-3 text-sm">
                              <select
                                className="input-field text-sm"
                                value={pending.level ?? u.level}
                                onChange={(e) => handleEditChange(u.id, 'level', e.target.value)}
                              >
                                {userLevels.map((level) => (
                                  <option key={level} value={level}>
                                    {level}
                                  </option>
                                ))}
                              </select>
                            </td>
                            <td className="px-4 py-3 text-sm">
                              <label className="inline-flex items-center gap-2">
                                <input
                                  type="checkbox"
                                  checked={pending.is_superuser ?? u.is_superuser}
                                  onChange={(e) => handleEditChange(u.id, 'is_superuser', e.target.checked)}
                                />
                                {pending.is_superuser ?? u.is_superuser ? 'Sim' : 'Não'}
                              </label>
                            </td>
                            <td className="px-4 py-3 text-sm">
                              <label className="inline-flex items-center gap-2">
                                <input
                                  type="checkbox"
                                  checked={pending.is_active ?? u.is_active}
                                  onChange={(e) => handleEditChange(u.id, 'is_active', e.target.checked)}
                                />
                                {pending.is_active ?? u.is_active ? 'Ativo' : 'Inativo'}
                              </label>
                            </td>
                            <td className="px-4 py-3 text-sm">
                              <button
                                onClick={() => handleSaveUser(u.id)}
                                disabled={!edits[u.id] || savingUserId === u.id}
                                className="btn-primary text-xs"
                              >
                                {savingUserId === u.id ? 'Salvando...' : 'Atualizar'}
                              </button>
                            </td>
                          </tr>
                        )
                      })}
                    </tbody>
                  </table>
                </div>
              )}
            </div>
          </section>
        )}

        {activeTab === 'repositories' && (
          <section className="card">
            <h3 className="font-semibold mb-4 text-gray-900">Gerenciamento de Repositórios</h3>
            <p className="text-gray-600">Funcionalidade de gerenciamento de repositórios será implementada em breve.</p>
          </section>
        )}

        {activeTab === 'proposals' && (
          <section className="card">
            <h3 className="font-semibold mb-4 text-gray-900">Gerenciamento de Propostas</h3>
            <p className="text-gray-600">Funcionalidade de gerenciamento de propostas será implementada em breve.</p>
          </section>
        )}

        {activeTab === 'issues' && (
          <section className="card">
            <h3 className="font-semibold mb-4 text-gray-900">Gerenciamento de Issues</h3>
            <p className="text-gray-600">Funcionalidade de gerenciamento de issues será implementada em breve.</p>
          </section>
        )}

        {activeTab === 'settings' && (
          <section className="card space-y-4">
            <h3 className="font-semibold text-gray-900">Configurações do Sistema</h3>
            <div className="space-y-3">
              <div className="flex items-center justify-between p-3 border border-gray-200 rounded">
                <div>
                  <p className="font-medium text-gray-900">Modo de manutenção</p>
                  <p className="text-sm text-gray-600">Desabilitar acesso ao sistema temporariamente</p>
                </div>
                <input type="checkbox" className="w-4 h-4" />
              </div>
              <div className="flex items-center justify-between p-3 border border-gray-200 rounded">
                <div>
                  <p className="font-medium text-gray-900">Registros de auditoria</p>
                  <p className="text-sm text-gray-600">Ativar logs de todas as ações administrativas</p>
                </div>
                <input type="checkbox" defaultChecked className="w-4 h-4" />
              </div>
              <div className="flex items-center justify-between p-3 border border-gray-200 rounded">
                <div>
                  <p className="font-medium text-gray-900">Permitir registros públicos</p>
                  <p className="text-sm text-gray-600">Permitir que novos usuários se registrem no sistema</p>
                </div>
                <input type="checkbox" defaultChecked className="w-4 h-4" />
              </div>
            </div>
            <button className="btn-primary mt-4">Salvar Configurações</button>
          </section>
        )}
      </div>
    </div>
  )
}

interface StatCardProps {
  icon: React.ReactNode
  value: number
  label: string
}

function StatCard({ icon, value, label }: StatCardProps) {
  return (
    <div className="card flex items-center gap-4">
      <div className="rounded-full bg-primary-100 text-primary-600 p-3">{icon}</div>
      <div>
        <div className="text-2xl font-bold text-gray-900">{value}</div>
        <div className="text-sm text-gray-500">{label}</div>
      </div>
    </div>
  )
}
